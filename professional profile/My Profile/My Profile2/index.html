SELECT
    c.customer_name,
    COUNT(o.order_id) AS total_orders,
    SUM(o.order_value) AS lifetime_value
FROM customers c
JOIN orders o
    ON c.customer_id = o.customer_id
GROUP BY c.customer_name
ORDER BY lifetime_value DESC;
